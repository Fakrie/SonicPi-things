#_Variables__________________________________________
rhythm = 1.80
amplitude = 0.5
High_amplitude= 1.5
Higher_amplitude= 2

#_Rhythm_____________________________
define :one do
  sleep rhythm
end
define :two do
  sleep rhythm*0.5
end
define :twon do
  sleep rhythm*0.5*1.5
end
define :four do
  sleep rhythm*0.25
end
define :fourn do
  sleep rhythm*0.25*1.5
end
define :eight do
  sleep rhythm*0.125
end
define :eightn do
  sleep rhythm*0.125*1.5
end
define :sxt do
  sleep rhythm*0.0625
end
define :sxtn do
  sleep rhythm*0.0625*1.5
end
define :thrt do
  sleep rhythm*0.03125
end

define :tremolo do |note1a,note1b,note2|
  8.times do
    play note1a
    play note1b
    sleep rhythm*0.03125
    play note2
    sleep rhythm*0.03125
  end
end
