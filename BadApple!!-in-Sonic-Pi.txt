define :drum do
  sample :bd_haus, amp: ampli

define :cymbal do
  sample :drum_cymbal_open,amp: ampli*0.55,rate: 1.1, attack: 0.01, sustain: 0, release: 0.2

  sample :drum_splash_hard, amp: ampli

define :abunchofnotes do |variable|
  s = play variable[0], amp: variable[1],attack: 0.1, sustain: 1, release: 0.01
  sleep 0.75
  control s, note: variable[0]+12
  sleep 0.5
  s = play variable[0]+12, amp: variable[1], sustain:  1, release: 0.01
  sleep 0.25
  control s, note: variable[0]+10
  sleep 0.25
  control s, note: variable[0]+12
  sleep 0.25
end
define :propdsaw do |variable|
  in_thread do
    if variable[0] == 0
      use_synth :prophet
      notes [prophetvolume, variable[1], variable[2]]
    else
      use_synth :dsaw
      notes [dsawvolume, variable[1], variable[2]]
    end
  end
end
define :bleyd do |variable|
  in_thread do
    use_synth :blade
    use_synth_defaults amp: bladevolume, cutoff: 120
    if variable[0]== 0
      blade = play :Eb4, sustain: variable[1]
      notes3 [blade, variable[2]]
    else
      if variable[0]==1
        blade = play :Db5, sustain: variable[1]
        notes4 [blade,variable[2],0,0]
        notes4 [blade,variable[2],variable[3],variable[4]]
      else
        blade = play :Db5, sustain: variable[1]
        notes4 [blade, variable[2],variable[3],variable[4]]
        notes2 [blade, variable[2]]
      end
    end
  end
end

define :notes do |variable|
  in_thread do
    4.times do
      3.times do
        abunchofnotes [variable[1],variable[0]]
      end
      cue :ths
      sync :thsfeedback
    end
  end
  2.times do |x|
    sync :ths
    s= play variable[1]+12, sustain: 2, amp: variable[0], attack: 0.1, release: 0.01
    5.times do |y|
      control s, note: variable[1]+notalar8[y]
      sleep zamanlama8[y]
    end
    cue :thsfeedback
    sync :ths
    if variable[2] == 1 && x == 1
      stop
    end
    s= play variable[1]+17, sustain: 2, amp: variable[0], attack: 0.1, release: 0.01
    6.times do |y|
      control s, note: variable[1]+notalar9[y]
      sleep zamanlama9[y]
    end
    cue :thsfeedback
  end
end

  in_thread do sus= 1
    use_synth :zawa
    play :Eb3,amp: ampli, attack: 0.1, release: 0.1, sustain: sus
    sleep 2
    7.times do |x|
      2.times do |y|
        if calcs[z][y]== 0
          sus = sus - suses[x]
        else
          sus = sus + suses[x]
        end
        play :Eb3,amp: ampli, attack: 0.1, release: 0.1, sustain: sus
        sleep 2
      end
      if z== 0
        z= z+1
      else
        z= z-1
      end
    end
  end
end
